N=int(input())

res=[]

    
for i in range(2,N+1):
    for j in range(2,i):
        if i%j==0:
            break
            
    else:
        res.append(i)
        
print(res)

print(len(res))
    


n=int(input())
ch=[0]*(n+1)
cnt=0

for i in range(2,n+1):
    if ch[i]==0:
        cnt+=1
        for j in range(i,n+1,i):
            ch[j]=1
print(cnt)
